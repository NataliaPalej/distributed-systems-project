CREATE TABLE IF NOT EXISTS trips (
	tripId INTEGER GENERATED BY DEFAULT AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY, 
	destination VARCHAR(100), 
	startDate VARCHAR(100),
	endDate VARCHAR(100), 
	budget DECIMAL(10,2),
	notes VARCHAR(255)
);

CREATE TABLE IF NOT EXISTS activities (
    activityId INTEGER GENERATED BY DEFAULT AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY, 
    tripId INTEGER NOT NULL,
    name VARCHAR(100), 
    activityDate VARCHAR(100),
    location VARCHAR(100), 
    cost DECIMAL(10,2) DEFAULT 0.00,
    FOREIGN KEY (tripId) REFERENCES trips(tripId) ON DELETE CASCADE
);

select * from activities;

select * from trips; 

DROP TABLE IF EXISTS activities;

DROP TABLE IF EXISTS trips;
